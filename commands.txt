# Hide mounted disk from Dock

$ gsettings set org.gnome.shell.extensions.dash-to-dock show-mounts false


# Show mounted disk from dock

$ gsettings set org.gnome.shell.extensions.dash-to-dock show-mounts true


# Hide user folder from Desktop

$ gsettings set org.gnome.shell.extensions.desktop-icons show-home false


# Show user folder from Desktop

$ gsettings set org.gnome.shell.extensions.desktop-icons show-home true


# Add 'Open in Terminator' in nautilus context menu

$ sudo apt install filemanager-actions
$ sudo apt install nautilus-actions

# Open Terminator in Desktop context menu

gsettings set com.gnome.desktop.default-applications.terminal exec /usr/bin/terminator

# Generate SSL in localhost

> Installation
$ sudo apt install libnss3-tools -y
$ wget https://github.com/FiloSottile/mkcert/releases/download/v1.1.2/mkcert-v1.1.2-linux-amd64
$ mv mkcert-v1.1.2-linux-amd64 mkcert
$ chmod +x mkcert
$ sudo cp mkcert /usr/local/bin/

> Generate Local CA
$ mkcert -install

> Generate Local SSL Certificates
$ mkcert localhost 127.0.0.1 ::1

> DNS clean
$ sudo /etc/init.d/dns-clean start

# Chnage default sort type to 'type'
$ gsettings set org.gnome.nautilus.preferences default-sort-order 'type'

# Clear memory
$ sudo sh -c 'echo 3 > /proc/sys/vm/drop_caches'
